15/// A feature-model formula file.
16///
17/// Every [Formula] may be parsed from an existing input [File].
18#[derive(Clone)]
19pub(crate) struct File {
20    /// The name of the file the associated formula was originally parsed from.
21    pub(crate) name: String,
22
23    /// The contents of the file the associated formula was originally parsed from.
24    pub(crate) contents: String,
25}
26
27impl File {
28    /// Creates a new file.
29    pub(crate) fn new(name: String, contents: String) -> Self {
30        Self { name, contents }
31    }
32
33    /// Returns whether a file exists at a given path.
34    ///
35    /// Also allows the special value - for referring to standard input.
36    pub(crate) fn exists(file_name: &str) -> bool {
37        Path::new(file_name).exists() || file_name.starts_with("-")
38    }
39
40    /// Reads the contents and extension of a file.
41    pub(crate) fn read(name: &str) -> File {
42        let mut contents;
43        if name.starts_with("-") {
44            contents = String::new();
45            std::io::stdin().read_to_string(&mut contents).unwrap();
46        } else {
47            contents = fs::read_to_string(name).unwrap();
48        };
49        File::new(name.to_string(), contents)
50    }
51
52    /// Writes this file to its destination.
53    pub(crate) fn write(&self) {
54        fs::write(&self.name, &self.contents).unwrap();
55    }
56
57    /// Returns the extension of this file, if any.
58    pub(crate) fn extension(&self) -> Option<String> {
59        Path::new(self.name.as_str())
60            .extension()
61            .map_or(None, |e| e.to_str())
62            .map(|e| e.to_string())
63    }
64
65    /// Counts the number of solutions of the formula this file represents using FeatureIDE.
66    ///
67    /// The file extension must be given so FeatureIDE can detect the correct format.
68    pub(crate) fn count_featureide(&self) -> BigInt {
69        exec::d4(&self.convert("cnf").contents)
70    }
71
72    /// Panics if the formula this file represents has a different model count than that of the given clauses.
73    ///
74    /// Useful for checking the correctness of count-preserving algorithms (e.g., [super::formula::Formula::to_cnf_tseitin]).
75    pub(crate) fn assert_count(&self, clauses: &Clauses) {
76        assert_eq!(clauses.count(), self.count_featureide());
77    }
78
79    /// Converts this file into a given format, if necessary.
80    pub(crate) fn convert(&self, output_format: &str) -> File {
81        if self
82            .extension()
83            .filter(|extension|  extension == output_format)
84            .is_none()
85        {
86            exec::io(&self, output_format, &[])
87        } else {
88            self.clone()
89        }
90    }
91
92    /// Slices the formula this file represents such that only the given variables remain.
93    ///
94    /// Internally, this uses FeatureIDE, so it operates on an intermediate CNF representation created by distributive transformation.
95    pub(crate) fn slice_featureide(
96        &self,
97        var_ids: &HashSet<VarId>,
98        arena: &mut Arena,
99        uvl: bool,
100    ) -> (Formula, Option<File>) {
101        let vars = var_ids
102            .iter()
103            .map(|var_id| {
104                let var_id: usize = var_id.unsigned_abs().try_into().unwrap();
105                if let Var::Named(name) = &arena.vars[var_id] {
106                    io::name_to_io(name)
107                } else {
108                    unreachable!()
109                }
110            })
111            .collect::<Vec<String>>();
112        let vars = vars.iter().map(|s| &**s).collect::<Vec<&str>>();
113        let slice = exec::io(&self, "cnf", &vars);
114        let mut uvl_file = None;
115        if uvl {
116            uvl_file = Some(exec::io(&self, "uvl", &vars));
117        }
118        let slice = Self::new("-.cnf".to_string(), io::name_from_io(&slice.contents));
119        let mut formula = arena.parse(slice, parser(Some("cnf".to_string())));
120        formula.sub_var_ids = var_ids.clone();
121        (formula, uvl_file)
122    }
123}
