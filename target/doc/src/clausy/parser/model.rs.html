15/// Parses feature-model formula files in the .model format.
16#[derive(Parser)]
17#[grammar = "parser/model.pest"]
18pub(crate) struct ModelFormulaParser;
19
20fn parse_children(
21    pair: Pair<Rule>,
22    arena: &mut Arena,
23    var_ids: &mut HashSet<VarId>,
24) -> Vec<ExprId> {
25    pair.into_inner()
26        .map(|pair| parse_pair(pair, arena, var_ids))
27        .collect()
28}
29
30fn parse_pair(pair: Pair<Rule>, arena: &mut Arena, var_ids: &mut HashSet<VarId>) -> ExprId {
31    match pair.as_rule() {
32        Rule::var => {
33            let (expr_id, var_id) = arena.var_expr_with_id(
34                pair.into_inner()
35                    .next()
36                    .unwrap()
37                    .as_str()
38                    .trim()
39                    .to_string(),
40            );
41            var_ids.insert(var_id);
42            expr_id
43        }
44        Rule::unsupported => {
45            let (expr_id, var_id) = arena.var_expr_with_id("<unsupported>".to_string());
46            var_ids.insert(var_id);
47            expr_id
48        }
49        Rule::not => {
50            let child_id = parse_pair(pair.into_inner().next().unwrap(), arena, var_ids);
51            arena.expr(Not(child_id))
52        }
53        Rule::and => {
54            let child_ids = parse_children(pair, arena, var_ids);
55            arena.expr(And(child_ids))
56        }
57        Rule::or => {
58            let child_ids = parse_children(pair, arena, var_ids);
59            arena.expr(Or(child_ids))
60        }
61        _ => unreachable!(),
62    }
63}
64
65fn parse_into(file: File, arena: &mut Arena) -> Formula {
66    let mut child_ids = Vec::<ExprId>::new();
67    let mut sub_var_ids = HashSet::<VarId>::new();
68    for line in file.contents.lines() {
69        let pair = ModelFormulaParser::parse(Rule::line, line)
70            .unwrap()
71            .next()
72            .unwrap();
73
74        match pair.as_rule() {
75            Rule::EOI => (),
76            _ => child_ids.push(parse_pair(pair, arena, &mut sub_var_ids)),
77        }
78    }
79    let root_id = arena.expr(And(child_ids));
80    Formula::new(sub_var_ids, root_id, Some(file))
81}
82
83impl FormulaParser for ModelFormulaParser {
84    fn parse_into(&self, file: File, arena: &mut Arena) -> Formula {
85        parse_into(file, arena)
86    }
