1//! Parser for inline input in a .sat-like format.
2
3use std::collections::HashSet;
4
5use crate::core::{
6    arena::Arena,
7    expr::{Expr::*, ExprId},
8    formula::Formula,
9};
10use pest::{iterators::Pair, Parser};
11use pest_derive::Parser;
12
13/// Parses inline input in a .sat-like format.
14///
15/// In this format, identifiers refer to previously parsed inputs.
16/// Optionally, the parsed formula can add negative backbone literals to align differing sets of variables.
17/// As sub-variables, the parsed formula has the union of the sub-variables of all given formulas.
18#[derive(Parser)]
19#[grammar = "parser/sat_inline.pest"]
20pub(crate) struct SatInlineFormulaParser<'a> {
21    formulas: &'a Vec<Formula>,
22    force_foreign_vars: Option<bool>,
23}
24
25impl<'a> SatInlineFormulaParser<'a> {
26    pub(crate) fn new(formulas: &'a Vec<Formula>, force_foreign_vars: Option<bool>) -> Self {
27        SatInlineFormulaParser {
28            formulas,
29            force_foreign_vars,
30        }
31    }
32
33    pub(crate) fn can_parse(file: &String) -> bool {
34        SatInlineFormulaParser::parse(Rule::file, file).is_ok()
35    }
36
37    pub(crate) fn parse_into(&self, file: &String, arena: &mut Arena) -> Formula {
38        let mut pairs = SatInlineFormulaParser::parse(Rule::file, file).unwrap();
39        let root_id = self.parse_pair(pairs.next().unwrap(), arena);
40        let sub_var_ids = if self.force_foreign_vars.is_some() {
41            arena.var_ids()
42        } else {
43            self.formulas
44                .iter()
45                .flat_map(|formula| formula.sub_var_ids.clone())
46                .collect()
47        };
48        Formula::new(sub_var_ids, root_id, None)
49    }
50
51    fn parse_children(&self, pair: Pair<Rule>, arena: &mut Arena) -> Vec<ExprId> {
52        pair.into_inner()
53            .map(|pair| self.parse_pair(pair, arena))
54            .collect()
55    }
56
57    fn parse_pair(&self, pair: Pair<Rule>, arena: &mut Arena) -> ExprId {
58        match pair.as_rule() {
59            Rule::var => {
60                let arg: i32 = pair
61                    .clone()
62                    .into_inner()
63                    .peek()
64                    .unwrap()
65                    .as_str()
66                    .parse()
67                    .unwrap();
68                let idx: usize = arg.try_into().unwrap();
69                let formula = &self.formulas[idx - 1];
70                let mut root_id = formula.root_id;
71                if self.force_foreign_vars.is_some() {
72                    root_id = formula
73                        .force_foreign_vars(self.force_foreign_vars.unwrap(), &HashSet::new(), arena)
74                        .root_id;
75                }
76                if pair.as_str().starts_with("-") {
77                    arena.expr(Not(root_id))
78                } else {
79                    root_id
80                }
81            }
82            Rule::not => {
83                let child_id = self.parse_pair(pair.into_inner().next().unwrap(), arena);
84                arena.expr(Not(child_id))
85            }
86            Rule::and => {
87                let child_ids = self.parse_children(pair, arena);
88                arena.expr(And(child_ids))
89            }
90            Rule::or => {
91                let child_ids = self.parse_children(pair, arena);
92                arena.expr(Or(child_ids))
93            }
94            _ => unreachable!(),
95        }
96    }
97}
