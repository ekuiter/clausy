14/// Returns the path of a bundled external program.
15///
16/// Looks up the program (a) as a sibling of the currently running executable, (b) in the working directory,
17/// and (c) in the `bin` directory in the working directory, if that exists.
18fn path(file_name: &str) -> String {
19    let mut path = env::current_exe().unwrap();
20    path.pop();
21    path.push(file_name);
22    if path.exists() {
23        return path.to_str().unwrap().to_owned();
24    }
25    let path = Path::new(file_name).to_path_buf();
26    if path.exists() {
27        return format!("./{}", file_name);
28    }
29    let path = Path::new(&format!("bin/{}", file_name)).to_path_buf();
30    if path.exists() {
31        return path.to_str().unwrap().to_owned();
32    }
33    unreachable!()
34}

36/// Attempts to find a solution of some CNF in DIMACS format.
37///
38/// Runs the external satisfiability solver counter kissat, which performs well on all known feature-model formulas.
39pub(crate) fn kissat(cnf: &str) -> Option<Vec<VarId>> {
40    let process = Command::new(path("kissat"))
41        .stdin(Stdio::piped())
42        .stdout(Stdio::piped())
43        .stderr(Stdio::piped())
44        .spawn()
45        .unwrap();
46    process.stdin.unwrap().write_all(cnf.as_bytes()).unwrap();
47    let mut output = String::new();
48    process.stdout.unwrap().read_to_string(&mut output).unwrap();
49    assert!(!output.is_empty());
50    let solution: Vec<VarId> = output
51        .lines()
52        .filter(|line| line.starts_with("v "))
53        .map(|line| line[2..].split(' ').collect::<Vec<&str>>())
54        .flatten()
55        .map(|str| str.parse().unwrap())
56        .filter(|literal| *literal != 0)
57        .collect();
58    if solution.len() > 0 {
59        Some(solution)
60    } else {
61        None
62    }
63}

65/// Counts the number of solutions of some CNF in DIMACS format.
66///
67/// Runs the external model counter d4, which performs well on most small to medium size inputs.
68/// Returns the number as a string, as it will typically overflow otherwise.
69pub(crate) fn d4(cnf: &str) -> BigInt {
70    let mut tmp = NamedTempFile::new().unwrap();
71    write!(tmp, "{}", cnf).unwrap();
72    let output = Command::new(path("d4"))
73        .arg("-i")
74        .arg(tmp.path())
75        .arg("-m")
76        .arg("counting")
77        .output()
78        .unwrap();
79    BigInt::from_str(&String::from(
80        String::from_utf8_lossy(&output.stdout)
81            .lines()
82            .find(|line| line.starts_with("s "))
83            .unwrap()
84            .split_at(2)
85            .1,
86    ))
87    .unwrap()
88}

90/// Enumerates all solutions of some CNF in DIMACS format.
91///
92/// Runs an external AllSAT solver, which is only suitable for formulas with few solutions.
93/// This does not currently output solutions for fully indeterminate (i.e., unconstrained) variables.
94pub(crate) fn bc_minisat_all(cnf: &str) -> (impl Iterator<Item = Vec<VarId>>, NamedTempFile) {
95    let mut tmp_in = NamedTempFile::new().unwrap();
96    write!(tmp_in, "{}", cnf).unwrap();
97    let process = Command::new(path("bc_minisat_all"))
98        .arg(tmp_in.path())
99        .stdin(Stdio::piped())
100        .stdout(Stdio::piped())
101        .stderr(Stdio::piped())
102        .spawn()
103        .unwrap();
104    let iter = BufReader::new(process.stderr.unwrap()).lines().map(|line| {
105        line.unwrap()
106            .split(' ')
107            .map(|literal| literal.parse().unwrap())
108            .filter(|literal| *literal != 0)
109            .collect::<Vec<VarId>>()
110    });
111    (iter, tmp_in)
112}

114/// Converts a given feature-model file from one format into another.
115///
116/// Runs the tool FeatureIDE using the Java runtime environment.
117pub(crate) fn io(
118    file: &File,
119    output_format: &str,
120    variables: &[&str],
121) -> File {
122    let process = Command::new("java")
123        .arg("-jar")
124        .arg(path("io.jar"))
125        .arg(&file.name)
126        .arg(output_format)
127        .arg(variables.join(","))
128        .stdin(Stdio::piped())
129        .stdout(Stdio::piped())
130        .stderr(Stdio::piped())
131        .spawn()
132        .unwrap();
133    if file.name.starts_with("-.") {
134        process.stdin.unwrap().write_all(file.contents.as_bytes()).unwrap();
135    }
136    let mut output = String::new();
137    let mut error = String::new();
138    process.stdout.unwrap().read_to_string(&mut output).unwrap();
139    process.stderr.unwrap().read_to_string(&mut error).unwrap();
140    if error.trim() == "No path set for model. Can't load imported models." {
141        error = String::new();
142    }
143    if !error.is_empty() {
144        println!("{}", error);
145    }
146    assert!(error.is_empty() && !output.is_empty());
147    File::new(format!("-.{}", output_format), output)
148}
